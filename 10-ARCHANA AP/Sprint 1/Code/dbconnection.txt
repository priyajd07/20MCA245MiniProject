import pymysql
def iud(querry,value):
    con=pymysql.connect(host="localhost",user="root",password="",port=3306,db="crime")
    cmd=con.cursor()
    cmd.execute(querry,value)
    id=cmd.lastrowid
    con.commit()
    con.close()
    return id

def androidselectall(q,val):
    con=pymysql.connect(host='localhost',port=3306,user='root',passwd='',db='crime')
    cmd=con.cursor()
    cmd.execute(q,val)
    s=cmd.fetchall()
    row_headers = [x[0] for x in cmd.description]
    json_data = []
    print(json_data)
    for result in s:
        json_data.append(dict(zip(row_headers, result)))
    return json_data
